Python Mid-conversation System Messages 
import anthropic

client = anthropic.Anthropic()

# 会話途中のシステムメッセージ (Mid-conversation System Messages)
response = client.messages.create(
    model="claude-opus-4-8-20260528",
    max_tokens=4096,
    system="あなたは優秀なソフトウェアエンジニアです。",
    messages=[
        {"role": "user", "content": "以下のコードをレビューしてください..."},
        {"role": "assistant", "content": "このコードには問題があります..."},
        # プロンプトキャッシュを維持したまま指示を動的に更新
        {"role": "system", "content": "今後はセキュリティの観点からもレビューしてください。"},
        {"role": "user", "content": "次のコードもレビューしてください..."}
    ]
)
