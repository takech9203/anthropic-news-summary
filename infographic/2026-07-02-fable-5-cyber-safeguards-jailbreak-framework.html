def calculate_cjs_level(capability_gain, breadth, ease, discoverability):
    """CJS レベルを算出"""
    if capability_gain == 0:
        return "CJS-0"
    total = capability_gain + breadth + ease + discoverability
    if total <= 3.5:
        return "CJS-1"
    elif total <= 6.5:
        return "CJS-2"
    elif total <= 8.5:
        return "CJS-3"
    else:
        return "CJS-4"

# 例: ユニバーサルプロンプトオーバーライド
print(calculate_cjs_level(4, 2, 2, 2))  # CJS-4
